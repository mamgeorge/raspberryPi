quick [

	https://www.instructables.com/id/Install-Nodejs-and-Npm-on-Raspberry-Pi/
	$ uname -m
	$ wget https://nodejs.org/dist/v8.9.0/node-v8.9.0-linux-armv6l.tar.gz
	$ mv node-v8.9.0-linux-armv6l.tar.gz nodejs.gz
	$ tar -xzf nodejs.gz
	$ cd nodejs/
	$ sudo cp -R * /usr/local/
	$ node -v
	$ npm -v
]
